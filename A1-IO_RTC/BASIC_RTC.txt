;       Requirements: PCB v0.94c, ATMEGA32 FW v0.94c
;       VIA Base Address = $2000 (8192 DECIMAL)
;       REQUIRES FOLLOWING ASM ROUTINES:
;       - SINGLE REGISTER READ
;       - SHIFT REGISTER DATA OUT

5 FOR I=0 TO 6				:REM READ EACH REGISTER FROM 0 TO 6: HOUR, MIN, SEC, DAY, MONTH, YEAR, RTC'S TEMPERATURE SENSOR
10 POKE 640,I				:SET REGISTER INDEX
20 CALL 1536				;CALL ROUTINE THAT READ THE REGISTER
30 V=PEEK(641):PRINT V;			;READ REGISTER'S VALUE AND PRINT IT
31 IF I<2 THEN PRINT ":";		;
33 IF (I>2 AND I<5) THEN PRINT "/";	;THESE LINES PUT SEPARATORS AMONG VALUES
34 IF I=5 THEN PRINT "   TEMP=";	;
40 NEXT I				;
45 PRINT
50 GOTO 5
